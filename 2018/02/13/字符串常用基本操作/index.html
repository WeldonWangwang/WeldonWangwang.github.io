int str2int(const char str)
{
int temp = 0;
const char ptr = str;

if (*str == '-' || *str == '+') {
    str++;
}
while (*str != 0) {
    if ((*str < '0') || (*str > '9')) {
        break;
    }

    temp = temp * 10 + (*str - '0');
    str++;
}

if (*ptr == '-') {
    temp = -temp;
}

return temp;

}
